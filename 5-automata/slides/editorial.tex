\begin{frame}{題解 - 第一回合}
  （假設都是第 $i$ 個隊員翻開第 $i$ 張牌）
  \begin{itemize}
    \item 只有兩個數字怎麼解？
    \item\onslide<2->{第一個人寫他的數字，假設是 3}
    \item\onslide<2->{第二個人把 3 乘以 10 ，加上他自己的，再除以 7}
  \end{itemize}
\end{frame}

\begin{frame}{題解 - 第一回合}
  \begin{itemize}
    \item 有很多個數字？
    \item 每個人都把白板上的數字乘以 10，加上自己的，再除以 7
  \end{itemize}
\end{frame}

\begin{frame}{題解 - 第一回合}
  例子：數字是 $341$
  \begin{itemize}
    \item 第一個人：寫 $3$
    \item 第二個人：寫 $34$ 除以 $7$ 的餘數 $= 6$
    \item 第三個人：答 $61$ 除以 $7$ 的餘數 $= 5$
  \end{itemize}
\end{frame}

\begin{frame}{題解 - 第一回合}
  例子：數字是 $341$
  \begin{itemize}
    \item 第一個人：寫 $3$
    \item 第二個人：寫 $34$ 除以 $7$ 的餘數 $= 6$
    \item 第三個人：答 $61$ 除以 $7$ 的餘數 $= 5$
  \end{itemize}
\end{frame}

\begin{frame}{題解 - 第一回合}
  怎麼混過對手：
  \begin{itemize}
    \item 打亂順序，可以先說好如果餘 $0$ 就寫 $5$、餘 $1$ 就寫 $2$...
    \item 還可以把 $7, 8, 9$ 一起放進來混淆視聽，例如讓 $6$ 跟 $7$ 同時代表餘 $0$
  \end{itemize}
\end{frame}

\begin{frame}{題解 - 第二回合}
  目標：
  \begin{itemize}
    \item 同時紀錄\textbf{目前答案}跟\textbf{目前進度}
  \end{itemize}
\end{frame}

\begin{frame}{題解 - 第二回合}
  目前進度可以分成下面三個：
  \begin{itemize}
    \item 1 號：前兩個是 \twemoji{red circle}\twemoji{black circle}，再一顆紅椰子就找到字串了
    \item 2 號：前一個是 \twemoji{red circle}，下一顆是黑椰子就還是符合「紅黑紅」
    \item 3 號：完全不符合「紅黑紅」（像是 \twemoji{black circle}\twemoji{black circle}）
  \end{itemize}
\end{frame}

\begin{frame}{題解 - 第二回合}
  根據現在的椰子顏色，可以從現在狀態決定要切換到哪個狀態
  \begin{itemize}
    \item 假設前一個是 \twemoji{red circle}，現在是 \twemoji{black circle}，那會從狀態 2 切換到狀態 1
    \item 假設前兩個是 \twemoji{red circle}\twemoji{black circle}，現在是 \twemoji{red circle}，除了答案多一之外，還會從狀態 1 切換到狀態 2
  \end{itemize}
\end{frame}

\begin{frame}{題解 - 第二回合}
  \begin{itemize}
    \item 有可能是 $0, 1, 2, 3, 4$ 五種答案
    \item 我們可以把「五種答案」跟「三種狀態」組合在一起
      \item 總共需要 $5 \times 3 = 15$ 個數字表達這兩個特徵
      \item 剛好可以用 $1$ 到 $15$ 表示！
      \item 有辦法用更少的數字，有興趣的隊員可以想想看！
  \end{itemize}
\end{frame}
