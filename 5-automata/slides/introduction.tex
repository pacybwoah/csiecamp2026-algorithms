\begin{frame}{遊戲說明}
  \begin{itemize}
    \item 講台前方有十張朝下的\textbf{卡牌}（椰子殼碎片）。
    \item 小隊各自分配 1-10 號，依序上臺：
      \item 每個隊員只能看\textbf{一張牌}的內容。
      \item 看完之後，可以在白板上留下一個 \textbf{0-9 的數字}。
      \item 每個隊員看得到上一個隊員在白板留的數字。
      \item 最後一個隊員要回答跟這些卡牌有關的問題，越準得分越高！
  \end{itemize}
\end{frame}

\begin{frame}{椰子傳書 - 範例}
  \begin{itemize}
    \item 範例關卡：%% punch, 270c, 1f590
      \begin{itemize}
        \item 有兩張牌，每張是剪刀、石頭、布的其中一個
        \item 胖椰、瘦椰的\textbf{目標}：回答是左邊營、右邊贏、還是平手。
      \end{itemize}
  \end{itemize}

\begin{frame}{椰子傳書 - 範例}
  \begin{itemize}
    \item 一個可行的\textbf{策略}：
      \begin{itemize}
        \item 胖椰跟瘦椰說好：0 代表剪刀、1 代表石頭、2 代表布。
        \item 胖椰先看左邊那張牌，看完他把代表的數字寫在白板上。
        \item 瘦椰看白板就知道左邊出什麼。
        \item 瘦椰再看右邊的牌，就知道問題的答案了！
      \end{itemize}
  \end{itemize}
\end{frame}

\begin{frame}{遊戲說明}
  \begin{itemize}
    \item 除了回答問題得分，也可以透過「猜另一隊回答什麼」得分。
    \item 看完卡片的隊員會進到第二間教室，第二間教室有實況轉播。
    \begin{itemize}
      \item 看得到牌的內容、其他人翻了什麼牌、在白板寫了什麼。
    \end{itemize}
    \item 在兩隊的 10 號隊員回答前，1-9 號隊員可以看到敵隊的戰術，你們要試著破解對方的戰術。
    \item 如果 1-9 號隊員猜對敵隊的 10 號隊員回答什麼，就可以得分。
  \end{itemize}
\end{frame}

\begin{frame}{椰子傳書}
  你們的目標是：
  \begin{itemize}
    \item 找出好的策略讓自己得分。
    \item 隱藏自己的策略，讓對方看不出來你在幹嘛。
    \item 同時，試著看破對方在幹嘛。
  \end{itemize}
\end{frame}

\begin{frame}{藏寶遊戲 - 第一回合}
  \begin{itemize}
    \item 第一回合中，卡牌是 \textbf{0-9} 的數字，十張卡牌組合成一個\textbf{十位數的數字}。
    \item 10 號成員要回答：「這個數字除以 7 的餘數是幾？」
  \end{itemize}
\end{frame}

\begin{frame}{藏寶遊戲 - 第一回合}
 範例：
  \begin{center}
    \twemoji[height=0.4in]{keycap: 2}\twemoji[height=0.4in]{keycap: 4}\twemoji[height=0.4in]{keycap: 7}\twemoji[height=0.4in]{keycap: 9}\twemoji[height=0.4in]{keycap: 9}\twemoji[height=0.4in]{keycap: 0}\twemoji[height=0.4in]{keycap: 2}\twemoji[height=0.4in]{keycap: 2}\twemoji[height=0.4in]{keycap: 4}\twemoji[height=0.4in]{keycap: 5} \newline
    \onslide<2->$2479902245$ 除以 $7 = 354271749$ 餘 $2 \Rightarrow$ 回答 $2$
  \end{center}
\end{frame}

\begin{frame}{藏寶遊戲 - 第二回合}
  \begin{itemize}
    \item 第二回合中，卡牌會是\textbf{紅色椰子}或是\textbf{藍色椰子}（綠色椰子被吃光了）
    \item 10 號成員要回答：「\twemoji{red circle}\twemoji{blue circle}\twemoji{red circle} 在卡牌裡面出現幾次？」
  \end{itemize}
\end{frame}
