\begin{frame}{遊戲說明}
  \begin{itemize}
    \item 講台前方有 12 張朝下的\textbf{卡牌}（椰子殼碎片）。
    \item 小隊各自分配 1 - 12 號，依序上臺：
      \item 每個隊員只能看\textbf{一張牌}的內容。
      \item 看完之後，可以在白板上留下一個 \textbf{0-15 的整數}。
      \item 每個隊員看得到上一個隊員在白板留的數字。
      \item 最後一個隊員不用看牌，他要回答跟這些卡牌有關的問題，越準得分越高。
  \end{itemize}
\end{frame}

\begin{frame}{椰子傳書 - 範例}
  \begin{itemize}
    \item 範例關卡：%% punch, 270c, 1f590

        \item 有兩張牌，每張是剪刀、石頭、布的其中一個
        \item 胖椰、瘦椰的\textbf{目標}：回答是左邊贏、右邊贏、還是平手。
      \end{itemize}
  \end{itemize}

\begin{frame}{椰子傳書 - 範例}
  \begin{itemize}
    \item 一個可行的\textbf{策略}：
      \begin{itemize}
        \item 胖椰跟瘦椰說好：0 代表剪刀、1 代表石頭、2 代表布。
        \item 胖椰先看左邊那張牌，看完他把代表的數字寫在白板上。
        \item 瘦椰看白板就知道左邊出什麼。
        \item 瘦椰再看右邊的牌，就知道問題的答案了！
      \end{itemize}
  \end{itemize}
\end{frame}

\begin{frame}{椰子傳書 - 第一回合}
  \begin{itemize}
    \item 第一回合中，卡牌是 \textbf{1-9} 的數字，十張卡牌組合成一個\textbf{十位數的數字}。
    \item 12 號小隊員要回答：「這個數字除以 7 的餘數是幾？」
  \end{itemize}
\end{frame}

\begin{frame}{椰子傳書 - 第一回合}
 範例：
  \begin{center}
    \twemoji[height=0.4in]{keycap: 2}\twemoji[height=0.4in]{keycap: 4}\twemoji[height=0.4in]{keycap: 7}\twemoji[height=0.4in]{keycap: 9}\twemoji[height=0.4in]{keycap: 9}\twemoji[height=0.4in]{keycap: 1}\twemoji[height=0.4in]{keycap: 2}\twemoji[height=0.4in]{keycap: 2}\twemoji[height=0.4in]{keycap: 4}\twemoji[height=0.4in]{keycap: 5} \newline
    \onslide<2->$2479912245$ 除以 $7 = 354273177$ 餘 $6 \Rightarrow$ 回答 $6$
  \end{center}
\end{frame}

\begin{frame}{椰子傳書 - 回合過程}
  \begin{itemize}
    \item 前面桌子上有兩隊的卡牌和白板
    \item 用撲克牌當卡牌，數字從 1 - 9
    \item 每隊有三分鐘討論戰術跟順序
    \begin{itemize}
      \item 順序要決定好喔
    \end{itemize}
  \end{itemize}
\end{frame}

\begin{frame}{椰子傳書 - 回合過程}
  \begin{itemize}
    \item 三分鐘結束後，請兩隊排成一列
    \item 每個人上前：
    \begin{itemize}
      \item 看一張牌跟白板的數字
      \item 把白板數字擦掉，寫上自己的
    \end{itemize}
  \end{itemize}
\end{frame}

\begin{frame}{椰子傳書 - 回合過程}
  \begin{itemize}
    \item 如果只有一隊答對，那一隊獲勝
    \item 如果兩隊都答對，那\textbf{最快的}那一隊獲勝
    \begin{itemize}
      \item \textbf{不能用計算機}，所以策略越簡單越好
    \end{itemize}
  \end{itemize}
\end{frame}

\begin{frame}{椰子傳書 - 第二回合}
  \begin{itemize}
    \item 第二回合中，卡牌會是\textbf{紅色椰子}或是\textbf{黑色椰子}（棕色椰子被吃光了）
    \item 12 號隊員要回答：「\twemoji{red circle}\twemoji{black circle}\twemoji{red circle} 在卡牌裡面出現幾次？」
  \end{itemize}
\end{frame}

\begin{frame}{椰子傳書 - 第二回合}
