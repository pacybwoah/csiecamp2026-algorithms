\begin{frame}{遊戲說明}
  \begin{itemize}
    \item 講台前方有 12 張朝下的\textbf{卡牌}（椰子殼碎片），上面寫 1-9 的數字。
    \item 12 張牌一字排開，組成一個 12 位數的數字。
    \item 小隊各自分配 1-12 號，依序上臺：
    \begin{itemize}
      \item 不夠的話請隊輔和隨輔一起玩 \twemoji{pleading face}
    \end{itemize}
      \item 每個隊員只能看\textbf{一張牌}的內容。
      \item 看完之後，可以在白板上留下一個 \textbf{非負整數}。
      \begin{itemize}
        \item 可以是任意非負整數，但數字越小越好
      \end{itemize}
      \item 每個隊員看得到上一個隊員在白板留的數字。
      \item 最後一個隊員看完牌後，要在白板上寫下「這個數字除以 7 的餘數是幾？」
  \end{itemize}
\end{frame}

\begin{frame}{椰子傳書 - 回合過程}
  \begin{itemize}
    \item 前面桌子上有兩隊的卡牌和白板
    \item 用撲克牌當卡牌，數字從 1 - 9
    \item 每隊有三分鐘討論戰術跟順序
    \begin{itemize}
      \item 順序要決定好喔
    \end{itemize}
  \end{itemize}
\end{frame}

\begin{frame}{椰子傳書 - 回合過程}
  \begin{itemize}
    \item 三分鐘結束後，請兩隊排成一列
    \item 每個人上前：
    \begin{itemize}
      \item 看一張牌跟白板的數字
      \item 把白板數字擦掉，寫上自己的
    \end{itemize}
    \item 目標是要在最快的時間內、只用最小的數字溝通，得到正確的答案
  \end{itemize}
\end{frame}

\begin{frame}{椰子傳書 - 回合過程}
  \begin{itemize}
    \item 如果只有一隊答對，那一隊獲勝
    \item 如果兩隊都答對，那我們會對每一隊計算\textbf{分數}：
    \begin{itemize}
      \item 假設 $x$ 是你們隊作答過程中白板下寫的最大數字，$t$ 是總共花的時間（秒）
      \item 那你們的分數會是 $\left\lceil \frac{x}{15} \right\rceil \cdot t$，$\left\lceil \text{ } \right\rceil$ 意思是向上取整
      \item 只要數字都在 $15$ 以內、又夠快，就可以拿到最好的分數
    \end{itemize}
    \item 分數比較低的隊伍獲勝！
  \end{itemize}
\end{frame}

\begin{frame}{椰子傳書 - 第二回合}
  \begin{itemize}
    \item 第二回合的設定跟第一回合一樣
    \begin{itemize}
      \item 大家要排成一列，每個人看一張牌、在白板寫一個數字
    \end{itemize}
    \item 差別在於\textbf{牌代表的意思不一樣}，現在牌只分成\textbf{紅色椰子}（紅色的牌 \twemoji{red circle}）跟\textbf{黑色椰子}（黑色的牌 \twemoji{black circle}）
    \begin{itemize}
      \item 棕色椰子都被吃光了
    \end{itemize}
    \item 12 號隊員要回答：「\twemoji{red circle}\twemoji{black circle}\twemoji{red circle} 在卡牌裡面出現幾次？」
  \end{itemize}
\end{frame}
